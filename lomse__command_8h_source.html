81 class DocCommand
82 {
83 protected:
84  string m_name; //displayable name for undo/redo actions display
85  string m_checkpoint; //checkpoint data
86  ImoId m_idChk; //id for target object in case of partial checkpoint
87  ImoId m_idRefresh; //id for cursor object, for k_refresh policy
88  string m_error;
89  uint_least16_t m_flags;
90 
91  enum {
92  k_reversible = 0x0001,
93  k_recordable = 0x0002,
94  k_target_set_in_constructor = 0x0004,
95  k_included_in_composite_cmd = 0x0008,
96  };
97 
98  DocCommand(const string& name)
99  : m_name(name)
100  , m_idChk(k_no_imoid)
101  , m_idRefresh(k_no_imoid)
102  , m_flags(0)
103  {
104  }
105 
106 public:
108  virtual ~DocCommand() {}
109 
112  enum ECmdCursorPolicy {
113  k_do_nothing=0,
114  k_update_after_deletion,
115  k_update_after_insertion,
116  k_refresh,
117  };
118 
121  virtual int get_cursor_update_policy() = 0;
122 
124  enum ECmdUndoPolicy {
125  k_undo_policy_full_checkpoint=0,
126  k_undo_policy_partial_checkpoint,
127  k_undo_policy_specific,
128  };
129 
132  virtual int get_undo_policy() = 0;
133 
136  enum ECmdSelectionPolicy {
137  k_sel_do_nothing=0,
138  k_sel_clear,
139  k_sel_command_specific,
140  };
141 
144  virtual int get_selection_update_policy() = 0;
145 
146  //information
153  inline std::string get_name() { return m_name; }
154 
158  inline bool is_reversible() { return m_flags & k_reversible; }
159 
169  inline bool is_recordable() { return (m_flags & k_recordable) != 0; }
170 
174  inline string get_error() { return m_error; }
175 
177  virtual bool is_composite() = 0;
178 
179 
180 //excluded from public API. Only for internal use.
182 public:
183  inline bool is_target_set_in_constructor() {
184  return (m_flags & k_target_set_in_constructor) != 0;
185  }
186  inline bool is_included_in_composite_cmd() {
187  return (m_flags & k_included_in_composite_cmd) != 0;
188  }
189 
190  virtual void update_selection(SelectionSet* UNUSED(pSelection)) {}
191  inline void set_final_cursor_pos(ImoId id) { m_idRefresh = id; }
192  inline ImoId get_final_cursor_pos() { return m_idRefresh; }
193 
194  //settings
195  inline void mark_as_included_in_composite_cmd() { m_flags |= k_included_in_composite_cmd; }
196 
197  //actions
198  virtual int set_target(Document* pDoc, DocCursor* pCursor, SelectionSet* pSelection)=0;
199  virtual int perform_action(Document* pDoc, DocCursor* pCursor)=0;
200  virtual void undo_action(Document* pDoc, DocCursor* pCursor);
202 
203 protected:
204  void create_checkpoint(Document* pDoc);
205  void log_forensic_data(Document* pDoc, DocCursor* pCursor);
206  void set_command_name(const string& name, ImoObj* pImo);
207  int validate_source(const string& source);
208  virtual void log_command(ofstream &logger);
209 
210 };
