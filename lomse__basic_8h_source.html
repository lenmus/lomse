195  Rectangle& intersection(const Rectangle& rect)
196  {
197  T x2 = right();
198  T y2 = bottom();
199 
200  if (x < rect.x)
201  x = rect.x;
202  if (y < rect.y)
203  y = rect.y;
204  if (x2 > rect.right())
205  x2 = rect.right();
206  if (y2 > rect.bottom())
207  y2 = rect.bottom();
208 
209  width = x2 - x;
210  height = y2 - y;
211 
212  if (width <= 0 || height <= 0)
213  {
214  width = 0;
215  height = 0;
216  }
217 
218  return *this;
219  }

228  Rectangle& Union(const Rectangle& rect)
229  {
230  // ignore empty rectangles: union with an empty rectangle shouldn't extend
231  // this one to (0, 0)
232  if ( !width || !height )
233  {
234  *this = rect;
235  }
236  else if ( rect.width && rect.height )
237  {
238  T x1 = min(x, rect.x);
239  T y1 = min(y, rect.y);
240  T y2 = max(y + height, rect.height + rect.y);
241  T x2 = max(x + width, rect.width + rect.x);
242 
243  x = x1;
244  y = y1;
245  width = x2 - x1;
246  height = y2 - y1;
247  }
248  //else: we're not empty and rect is empty
249 
250  return *this;
251  }

343 inline URect normalized_rectangle(LUnits xLeft, LUnits yTop,
344  LUnits xRight, LUnits yBottom)
345 {
346  LUnits xL, xR, yT, yB;
347 
348  if (xLeft > xRight)
349  {
350  xL = xRight;
351  xR = xLeft;
352  }
353  else
354  {
355  xL = xLeft;
356  xR = xRight;
357  }
358 
359  if (yTop > yBottom)
360  {
361  yT = yBottom;
362  yB = yTop;
363  }
364  else
365  {
366  yT = yTop;
367  yB = yBottom;
368  }
369 
370  return URect(xL, yT, xR-xL, yB-yT);
371 }
